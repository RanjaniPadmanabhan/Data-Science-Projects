def el(x):
    if x<85:
        return 'green'
    if (x>=85) & (x<=98):
        return 'orange'
    if x>98:
        return 'red'
    
def ect(x):
    if x<90:
        return 'green'
    if (x>=90) & (x<=105):
        return 'orange'
    if x>105:
        return 'red'
    
def erpm(x):
    if x<1800:
        return 'green'
    if (x>=1800) & (x<=2200):
        return 'orange'
    if x>2200:
        return 'red'
    
def vcm(x):
    if (x>13.9) & (x<14.9):
        return 'green'
    if (x>=13) & (x<=13.9):
        return 'orange'
    if (x<13) | (x>=14.9):
        return 'red'
    
def eot(x):
    if (x<100):
        return 'green'
    if (x>=100) & (x<120):
        return 'orange'
    if (x>=120):
        return 'red'
    
def tb_vcm(x):
    if x<22:
        return 'green'
    if x>=22:
        return 'red'

def build_evaluate_fn(x_train, y_train, x_test, y_test):
    def evaluate(model):
        train_score = model.score(x_train, y_train)
        test_score = model.score(x_test, y_test)

        print("Train Score:", train_score)
        print("Test Score:", test_score)
        print()
        
        
    
    return evaluate
